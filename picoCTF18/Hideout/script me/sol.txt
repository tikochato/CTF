def nivel(n):
        nivel = 0
        nivelT = 0
        for a in n:
                if a == '(':
                        nivelT = nivelT +1
                else:
                        if nivelT > nivel:
                                nivel = nivelT
                        nivelT = nivelT -1
        return nivel
